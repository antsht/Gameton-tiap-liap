🧠 1. ЧТО ДОЛЖЕН ОЦЕНИВАТЬ AI

Каждый ход AI должен выбрать действие с максимальным:

TOTAL_SCORE = Immediate + Future - Risk - Inefficiency
Разбивка:
✅ Immediate (мгновенные очки)
терраформация
диверсия (если добивание)
бобры
🔮 Future (будущие очки)
новая плантация
выгодная позиция
⚠️ Risk
потеря ЦУ
изоляция
урон
📉 Inefficiency
потеря силы через цепочку
перегруз одной точки
🔥 2. ОСНОВНАЯ ФОРМУЛА
ActionScore =
    W1 * ImmediateScore +
    W2 * FutureScore -
    W3 * RiskScore -
    W4 * EfficiencyLoss
Рекомендуемые веса:
W1 = 1.0
W2 = 0.8
W3 = 1.5
W4 = 1.2

👉 Риск важнее прибыли — это ключ к стабильному топ-1

🧮 3. ДЕТАЛИЗАЦИЯ
🟢 3.1 Строительство
BuildScore =
    (CellValue * TS * ExpectedLifetime)
    - BuildTimePenalty
    - DistancePenalty
Где:
CellValue = 10 или 15
ExpectedLifetime ≈ 80–120 ходов
BuildTimePenalty = 50 - CS_effective
DistancePenalty = hops * 10
🔴 3.2 Диверсия
SabotageScore =
    KillBonus +
    DenyEnemyFuture -
    TravelPenalty
Где:
KillBonus = CellValue * RemainingProgress
DenyEnemyFuture ≈ 500–1500
🟣 3.3 Бобры
BeaverScore =
    10 * CellValue * CompletionFactor
    - CompetitionRisk
🔵 3.4 Ремонт
RepairScore =
    PreventedLoss * 1.5
⚠️ 3.5 РИСК
RiskScore =
    CU_Danger * 10000 +
    IsolationRisk * 2000 +
    LowHP * 500
📉 3.6 Потери эффективности
EfficiencyLoss = hops * 20 + overload * 50

🧮 4. НОВАЯ SCORING-МОДЕЛЬ (КЛЮЧЕВАЯ)
💥 Основная идея:
TotalScore = Σ Score(command_i)

НО:

Score(command_i) зависит от других команд

👉 это combinatorial optimization

⚙️ 5. ПРАВИЛЬНАЯ МОДЕЛЬ
Шаг 1 — генерируем цели
targets:
- build cells
- repair targets
- enemy targets
- beavers
Шаг 2 — считаем базовый score цели
TargetScore =
    CellValue * 100
    - DistancePenalty
    - Risk
Шаг 3 — считаем маржинальную пользу каждой команды

Вот где магия:

MarginalGain(n) = gain от n-й команды на цель
❗ ВАЖНО

Из-за механики:

каждая следующая команда → эффективность падает

👉 значит:

MarginalGain убывает
📉 Пример:
Команда	Польза
1	100
2	80
3	60
4	30
5	10

👉 значит:

❗ выгоднее распределять, а не спамить

💻 9. ГОТОВЫЙ C# АЛГОРИТМ
public class CommandAllocator
{
    public List<Command> Allocate(List<Target> targets, int totalCommands)
    {
        var result = new List<Command>();

        for (int i = 0; i < totalCommands; i++)
        {
            Target best = null;
            double bestScore = double.MinValue;

            foreach (var target in targets)
            {
                double score = target.GetMarginalGain();

                if (score > bestScore)
                {
                    bestScore = score;
                    best = target;
                }
            }

            if (best == null) break;

            result.Add(best.CreateCommand());
            best.ApplyCommand(); // уменьшает эффективность
        }

        return result;
    }
}

public class Target
{
    public int UsageCount;
    public int BaseValue;
    public int Distance;

    public double GetMarginalGain()
    {
        double efficiency = Math.Max(0, 5 - UsageCount - Distance);
        return BaseValue * efficiency;
    }

    public void ApplyCommand()
    {
        UsageCount++;
    }

    public Command CreateCommand()
    {
        return new Command();
    }
}

public class SimState
{
    public int Turn;
    public List<SimPlantation> Plants = new();
    public List<SimConstruction> Constructions = new();
    public double Score;

    public SimState Clone()
    {
        return new SimState
        {
            Turn = Turn,
            Score = Score,
            Plants = Plants.Select(p => p.Clone()).ToList(),
            Constructions = Constructions.Select(c => c.Clone()).ToList()
        };
    }
}

🔹 Симуляция хода
public static class Simulator
{
    public static SimState Simulate(SimState state, List<SimAction> actions)
    {
        var next = state.Clone();
        next.Turn++;

        // 1. Применяем команды
        foreach (var action in actions)
        {
            if (action.Type == ActionType.Build)
            {
                var c = next.Constructions
                    .FirstOrDefault(x => x.X == action.X && x.Y == action.Y);

                if (c == null)
                {
                    c = new SimConstruction { X = action.X, Y = action.Y };
                    next.Constructions.Add(c);
                }

                c.Progress += action.Power;
            }
        }

        // 2. Завершение стройки
        foreach (var c in next.Constructions.ToList())
        {
            if (c.Progress >= 50)
            {
                next.Plants.Add(new SimPlantation { X = c.X, Y = c.Y });
                next.Constructions.Remove(c);
            }
        }

        // 3. Терраформация
        foreach (var p in next.Plants)
        {
            int cellValue = ((p.X % 7 == 0) && (p.Y % 7 == 0)) ? 15 : 10;
            next.Score += p.TS * cellValue;
        }

        return next;
    }
}
🔹 Генерация действий
public class SimAction
{
    public ActionType Type;
    public int X;
    public int Y;
    public int Power;
}

public static class ActionGenerator
{
    public static List<List<SimAction>> Generate(SimState state)
    {
        var result = new List<List<SimAction>>();

        var dirs = new (int dx, int dy)[]
        {
            (1,0),(-1,0),(0,1),(0,-1)
        };

        foreach (var d in dirs)
        {
            var actions = new List<SimAction>();

            foreach (var p in state.Plants)
            {
                actions.Add(new SimAction
                {
                    Type = ActionType.Build,
                    X = p.X + d.dx,
                    Y = p.Y + d.dy,
                    Power = 1
                });
            }

            result.Add(actions);
        }

        return result;
    }
}
🔹 Beam Search
public class Planner
{
    public List<SimAction> FindBest(SimState root)
    {
        int depth = 3;
        int beamWidth = 5;

        var states = new List<(SimState state, List<SimAction> firstActions)>
        {
            (root, new List<SimAction>())
        };

        for (int d = 0; d < depth; d++)
        {
            var nextStates = new List<(SimState, List<SimAction>)>();

            foreach (var (state, first) in states)
            {
                var actionSets = ActionGenerator.Generate(state);

                foreach (var actions in actionSets)
                {
                    var next = Simulator.Simulate(state, actions);

                    var firstActions = first.Count == 0 ? actions : first;

                    nextStates.Add((next, firstActions));
                }
            }

            states = nextStates
                .OrderByDescending(s => s.Item1.Score)
                .Take(beamWidth)
                .ToList();
        }

        return states.First().firstActions;
    }
}
