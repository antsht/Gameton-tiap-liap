package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"net/http"
	"time"
)

const API = "https://games-test.datsteam.dev/api"
const TOKEN = "YOUR_TOKEN_HERE"

func main() {
	client := &http.Client{Timeout: 500 * time.Millisecond}

	for {
		state := GetState(client)
		if state == nil {
			continue
		}

		commands := Plan(state)

		SendCommand(client, commands)

		time.Sleep(200 * time.Millisecond)
	}
}

func GetState(client *http.Client) *GameState {
	req, _ := http.NewRequest("GET", API+"/arena", nil)
	req.Header.Set("X-Auth-Token", TOKEN)

	resp, err := client.Do(req)
	if err != nil {
		return nil
	}
	defer resp.Body.Close()

	var state GameState
	json.NewDecoder(resp.Body).Decode(&state)
	return &state
}

func SendCommand(client *http.Client, cmds []Command) {
	payload := map[string]interface{}{
		"command": cmds,
	}

	data, _ := json.Marshal(payload)

	req, _ := http.NewRequest("POST", API+"/command", bytes.NewBuffer(data))
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("X-Auth-Token", TOKEN)

	client.Do(req)
}

🎯 4. TARGET SYSTEM
type TargetType int

const (
	Build TargetType = iota
	Sabotage
	Repair
)

type Target struct {
	X, Y   int
	Type   TargetType
	Value  float64
	Usage  int
}

func (t *Target) MarginalGain(dist int) float64 {
	eff := float64(5 - t.Usage - dist)
	if eff < 0 {
		eff = 0
	}
	return t.Value * eff
}
⚙️ 5. GENERATE TARGETS
func GenerateTargets(state *GameState) []Target {
	var targets []Target

	dirs := [][2]int{{1,0},{-1,0},{0,1},{0,-1}}

	for _, p := range state.Plantations {
		x, y := p.Position[0], p.Position[1]

		for _, d := range dirs {
			nx, ny := x+d[0], y+d[1]

			val := 1000.0
			if nx%7 == 0 && ny%7 == 0 {
				val = 1500.0
			}

			targets = append(targets, Target{
				X: nx, Y: ny,
				Type: Build,
				Value: val,
			})
		}
	}

	for _, e := range state.Enemy {
		targets = append(targets, Target{
			X: e.Position[0],
			Y: e.Position[1],
			Type: Sabotage,
			Value: 1200,
		})
	}

	return targets
}
🧮 6. COMMAND ALLOCATION (CORE)
func Allocate(state *GameState, targets []Target) []Command {
	total := len(state.Plantations) * 3
	var result []Command

	for i := 0; i < total; i++ {
		bestIdx := -1
		bestScore := -1.0

		for j := range targets {
			score := targets[j].MarginalGain(1)

			if score > bestScore {
				bestScore = score
				bestIdx = j
			}
		}

		if bestIdx == -1 {
			break
		}

		targets[bestIdx].Usage++

		cmd := Command{
			Path: [][]int{
				state.Plantations[0].Position[:],
				{targets[bestIdx].X, targets[bestIdx].Y},
			},
		}

		result = append(result, cmd)
	}

	return result
}
🔮 7. MULTI-TURN (BEAM SEARCH)
func Plan(state *GameState) []Command {
	targets := GenerateTargets(state)

	type Node struct {
		State *SimState
		Cmds  []Command
		Score float64
	}

	root := Node{
		State: ConvertToSim(state),
	}

	beam := []Node{root}

	for depth := 0; depth < 3; depth++ {
		var next []Node

		for _, n := range beam {
			candidates := GenerateActionSets(n.State)

			for _, c := range candidates {
				s2 := Simulate(n.State, c)
				score := EvaluateSim(s2)

				cmds := n.Cmds
				if depth == 0 {
					cmds = c
				}

				next = append(next, Node{
					State: s2,
					Cmds:  cmds,
					Score: score,
				})
			}
		}

		beam = TopK(next, 6)
	}

	return beam[0].Cmds
}
🧠 8. SIMULATOR (упрощённый, но ключевой)
type SimState struct {
	Plants [][2]int
	Score  float64
}

func ConvertToSim(state *GameState) *SimState {
	var plants [][2]int
	for _, p := range state.Plantations {
		plants = append(plants, p.Position)
	}
	return &SimState{Plants: plants}
}

func Simulate(s *SimState, cmds []Command) *SimState {
	next := *s

	for _, p := range s.Plants {
		val := 10.0
		if p[0]%7 == 0 && p[1]%7 == 0 {
			val = 15
		}
		next.Score += val * 5
	}

	// ускорение роста (имитация строительства)
	if len(cmds) > 3 {
		next.Plants = append(next.Plants, [2]int{s.Plants[0][0]+1, s.Plants[0][1]})
	}

	return &next
}
🧮 9. EVALUATION (ключ к топ-1)
func EvaluateSim(s *SimState) float64 {
	score := s.Score

	// бонус за количество плантаций
	score += float64(len(s.Plants)) * 200

	// штраф за разреженность
	if len(s.Plants) > 0 {
		score -= float64(len(s.Plants)) * 5
	}

	return score
}
🔝 10. TOP-K
func TopK(nodes []Node, k int) []Node {
	sort.Slice(nodes, func(i, j int) bool {
		return nodes[i].Score > nodes[j].Score
	})

	if len(nodes) < k {
		return nodes
	}
	return nodes[:k]
}
